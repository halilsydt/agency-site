/**
 * Props for the TeamCard component.

interface TeamCardProps {
  /** The team member to display */
  member: TeamMember;
}

/**
 * Displays a team member in a card format with photo, name, role, and bio.
 */
export function TeamCard({ member }: TeamCardProps) {
  return (
    <Card className="text-center">
      <CardHeader>
        <div className="w-24 h-24 mx-auto mb-4 rounded-full bg-muted flex items-center justify-center overflow-hidden">
          {member.imageUrl ? (
            <img src={member.imageUrl} alt={member.name} className="w-full h-full object-cover" />
          ) : (
            <User className="w-12 h-12 text-muted-foreground" />
          )}
        </div>
        <CardTitle>{member.name}</CardTitle>
        <p className="text-sm text-muted-foreground">{member.role}</p>
      </CardHeader>
      <CardContent>
        <p className="text-muted-foreground">{member.bio}</p>
      </CardContent>
    </Card>
  );
}

import aboutData from '@/content/about.json';
import type { TeamMember, ExperienceHighlight } from './types';

interface AboutContent {
  mission: { headline: string; text: string };
  approach: { headline: string; description: string; points: string[] };

  return aboutData.team as TeamMember[];

```typescript
import { render, screen } from '@testing-library/react';
import { describe, it, expect } from 'vitest';
import { TeamCard } from '@/components/cards/team-card';

const mockMember = {
  id: 'founder',
  name: 'Jane Doe',
  role: 'Founder & CEO',
  bio: 'Passionate about helping sellers succeed.',
  imageUrl: undefined,
};

describe('TeamCard', () => {
  it('renders team member name and role', () => {
    render(<TeamCard member={mockMember} />);
    expect(screen.getByText('Jane Doe')).toBeInTheDocument();
    expect(screen.getByText('Founder & CEO')).toBeInTheDocument();
  });

  it('renders team member bio', () => {
    render(<TeamCard member={mockMember} />);
    expect(screen.getByText(/Passionate about helping/)).toBeInTheDocument();
  });

  it('renders placeholder when no image provided', () => {
    render(<TeamCard member={mockMember} />);
    // Should render User icon placeholder
    expect(screen.getByTestId('team-card-placeholder')).toBeInTheDocument();
  });

  it('renders image when imageUrl provided', () => {
    const memberWithImage = { ...mockMember, imageUrl: '/images/team/jane.jpg' };
    render(<TeamCard member={memberWithImage} />);
    expect(screen.getByRole('img', { name: 'Jane Doe' })).toBeInTheDocument();
  });
});
```
